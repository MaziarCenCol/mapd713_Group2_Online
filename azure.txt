Deployment:
1-Download profile from azure
2-in GitHub: setting->secret and ... -> new secret -> paste the content from downloaded file
3-in VS code .GitHub/workflows/deploy.yml
name: Deploy to Azure App Service

on:
  push:
    branches:
      - main

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout source
        uses: actions/checkout@v3

      - name: Setup Node.js version
        uses: actions/setup-node@v4
        with:
          node-version: "18.x"

      - name: Install dependencies
        run: npm install

      - name: Deploy to Azure App Service
        uses: azure/webapps-deploy@v2
        with:
          app-name: mapd713nodeapi
          publish-profile: ${{ secrets.MAPD713_GROUP2_SECRET }}
          package: .
          
(replace mapd713nodeapi and MAPD713_GROUP2_SECRET)

see in Action the success tag.
